load_json_example <- function(file, dir = "modified-hubmeta-examples") {
    jsonlite::read_json(
    here::here("json-schema", dir, file),
    simplifyVector = TRUE, 
    simplifyDataFrame = FALSE)
}

apply_validator <- function(path) {
    validated <- validator(path, verbose = TRUE) 
    if (validated) {
        validated
    } else {
        validated |>
            attr("errors") |> 
            View()
    }
}
